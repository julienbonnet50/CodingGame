    public void addCard(Card cardToAdd) {

            cards = new Card[]{cardToAdd}; // Création d'un nouveau tableau pour une seule carte

        } else {

            int currentSize = cards.length;

            Card[] newHand = new Card[currentSize + 1];


            // Copie des cartes actuelles dans le nouveau tableau

            System.arraycopy(cards, 0, newHand, 0, currentSize);


            // Ajout de la nouvelle carte à la fin du tableau

            // Remplacement du tableau de cartes par le nouveau

            cards = newHand;

            }

        }
